git status - проверка статуса репозитория
git add 'путь к файлу' - индексация одного файла
git add . - индексация всех изменений
git commit -m "комментарий" - коммит проиндексированных изменений///
git checkout "path" - отмена локальных изменений для одного файла
git checkout -- .  - отмена локальных изменений для всех файлов
git reset HEAD "path" - отмена проиндексированных изменений
git reset HEAD -- .  - отмена проиндексированных изменений для всех файлов
